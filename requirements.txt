# Separación de fuentes de audio. Instala automáticamente PyTorch como
# dependencia (puede pesar varios cientos de MB a un par de GB según tu

# que se ejecuta (requiere internet en ese momento).
demucs

# Lectura/escritura de archivos de audio
soundfile
numpy

# Monitoreo de recursos (CPU, RAM, threads del proceso)
psutil

# Paralelismo multiproceso en un solo computador (modo "joblib"/"parallel")
joblib

# Paralelismo/distribución de tareas (modo "ray")
ray[default]

# Generación de los gráficos comparativos
matplotlib

# Opcional pero recomendado si hay GPU NVIDIA disponible: permite medir

# sin fallar.
nvidia-ml-py
